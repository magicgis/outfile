Type in the text of a new option tag to add dynamically.

$("select").multiselect({
	   header: "Select an option",
	   url:'/jcsjfxpt/test/multiselectTest.do',
	   classes:'select',
	   data:[{'text':'飞','value':'9'},{'text':'发','value':'10'},{'text':'放','value':'13'}],
	   executeMessage:"加载中...",
	   successMessage:'数据加载完成'
	}).multiselectfilter({//设置过滤器属性
		label:'主键:'
	});
	 var el = $("select").multiselect(),
		disabled = $('#disabled'),
		selected = $('#selected'),
		newItem = $('#newItem');
	
	$("#add").click(function(){
		var v = newItem.val(), opt = $('<option />', {
			value: v,
			text: v
		});
		
		if(disabled.is(':checked')){
			opt.attr('disabled','disabled');
		};
		if(selected.is(':checked')){
			opt.attr('selected','selected');
		};
		
		opt.appendTo( el );
		el.multiselect('refresh');
	});
		
	var flag=false;
	$("#del").click(function(){
		var enable=flag?'enable':'disable';
		$("select").multiselect(enable);
		flag=!flag;
		
	});
